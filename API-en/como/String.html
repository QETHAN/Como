trim escapeHTML unescapeHTML removeHTML byteLength delLast toInt toArray left right format toLower toUpper on16 un16
trim
remove space of head and end
' Hello World!     '.trim();			//return 'Hello World!'
escapeHTML
Escape HTML
'<b>test</b>'.escapeHTML();			//return '&lt;b&gt;test&lt;/b&gt;'
unescapeHTML
be contrary to escapeHTML
'&lt;b&gt;test&lt;/b&gt;'.unescapeHTML();		//return  '<b>test</b>'
removeHTML
remove HTML tags 
'<b>test</b>'.removeHTML();		//return 'test'
byteLength
get the string's length, a Chinese char is two lengths
'abc你好'.byteLength();		//return 7
delLast
remove the last char
'abc'.delLast();		//return 'ab'
toInt
String to Int
"111".toInt();		//return 111
"aa11".toInt();		//return NaN
toArray
String to Array
"111".toArray();		//return ["1", "1", "1"]
"111".toArray('');		//return ["1", "1", "1"]
"1,1,1".toArray(',');		//return ["1", "1", "1"]
left
substring, start from head, and a Chinese char is two lengths
"test".left(2);			//return "te"
right
substring, start from end, and a Chinese char is two lengths
"test".right(2);		//return "st"
format
format string
"test{0}test{1}".format("A", "B");		//return "testAtestB"
toLower
toLowerCase
"test".toLower();		//return "test"
toUpper
toUpperCase
"test".toUpper();		//return "TEST"
on16
toString(16)
"test".on16();		//return "\u0074\u0065\u0073\u0074"
un16
toString(10)
"\u0074\u0065\u0073\u0074".un16();		//return "test"
