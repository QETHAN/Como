Selector
CSS Selector, return an array of Dom Element(Como-Dom), just like JQuery.
Attention, if there isn't Element, return null.
We can understand this: $("#test")[0] = document.getElementById("test"); $("#test") = [document.getElementById("test")];
With the Only mode, user Como, not $.

About CSS Selector, Como support some common syntax, such as id, tag, class, child, attribute.
Actually you can use these syntax to solve all the problems.
a test of selector speed about Como contrast to other common frameworks.
The following demo: 
$("#id")					//by id
$(".class")					//by class name
$("div.class")					//by tag and class name
$("tag")					//by tag
$(".test #test1")
$(".test div")					//
$(".test>div")					//
$(".test a[attribute]")				//by attribute
$(".test a[attribute=value]")			//by attribute and attribute's value
$("div div")
$(".test>div.class")

como object's method
size get each index attr prop remove css text html val append prepend before after down up upWithMe on un out unout simulate left top pos height width show hide toggle focus prev prevAll next nextAll first last children parent hasClass addClass removeClass removeAttr removeCSS anim
size
get Como-Dom's length
$("span").size();
get
get the index item of Como-Domï¼Œreturn Como-Dom
$("span").get(1);		//get the second item of Como-Dom
each
loop Como-Dom and setup
$("span").each(function(el){
	alert(el.html());
});
index
get the index that gived Dom or Como in Como-Dom
$("span").index($("#test1"));
$("span").index(document.getElementById("test1"));
attr
get or set the value of Dom's attribute
$("#test").attr('class');
$("#test").attr('class', 'className');
prop
get or set the value of Dom's prototype
$("#test").prop('className');
$("#test").prop('className', 'className');
remove
remove Dom
$("#test").remove();
css
get or set the style value of Dom
$("#test").css('display');
$("#test").css('display', 'none');
$("#test").css('opacity');
$("#test").css('opacity', 0.5)
text
get or set the innerText of Dom
$("#test").text();
$("#test").text("Hello World!")
html
get or set the innerHTML of Dom
$("#test").html();
$("#test").html("Hello World!")
val
get or set the value of form Element
$("input").val();
$("input").val('Hello');
$("radio").val();
$("checkbox").val();
append
append Element or String at the end of Dom
$("#test").append("<span>Hello World!</span>");
prepend
append Element or String at the start of Dom
$("#test").prepend("<span>Hello World!</span>");
before
append Element or String before the Dom
$("#test").before("<span>Hello World!</span>");
after
append Element or String after the Dom
$("#test").after("<span>Hello World!</span>");
down
select dom in the Dom's children, like searching down
$("#test").down("div");
up
select dom in those who are parent of the Dom, like searching up
$("#test").up("div");
upWithMe
select dom in those who are parent of the Dom or is the Dom, like searching up;
with up(), the difference is that upWithMe() contains the Dom itself.
$("#test").upWithMe("div");
on
attach a handler to an event for the Dom, like bind() in JQuery.
$("#test").on("click", function(){});
un
cancel the handler which had bind.
$("#test").un("click", function(){});
//the function that on() and un() must be the same pointer.
out
bind event which will be fired out of the Dom.
$("#test").out("click", function(){});		//when click out of the id=test, then event fired.
$("#test").out("click", function(){}, true);	//when click out of the id=test, then event fired, and cancel the bind
unout
cancel the out()

	

	

		
		
	


	
	
simulate

	
simulate user to fire event

	


$('#test').simulate('click');  //fire the click event at id=test

	


	
	
left

	
get or set the Dom's X position

	


$("#test").left();
$("#test").left(100)

	

	

		
		
	


	
	
top

	
get or set the Dom's Y position

	


$("#test").top();
$("#test").top(100);

	

	

		
		
	


	
	
pos

	
get the Dom's position, return {left: 500, top: 500}

	


$("#test").pos()

	

	

		
		
	


	
	
height

	
get or set the Dom's height

	


$("#test").height()
$("#test").height(500)

	

	

		
		
	


	
	
width

	
get or set the Dom's width

	


$("#test").width();
$("#test").width(500)

	

	

		
		
	


	
	
show

	
show the Dom.

	


$("#test").show()

	

	

		
		
	


	
	
hide

	
hide the Dom.

	


$("#test").hide()

	

	

		
		
	


	
	
toggle

	
if the Dom is hiding, then show it, or hide it.

	


$("#test").toggle()

	

	

		
		
	


	
	
focus

	
set the form element on focus.

	


$("#input1").focus()

	

	

		
		
	


	
	
prev

	
get the index item before the Dom, index's default value is 0.

	


$("#test").prev();
$("#test").prev(2);

	

	

		
		
	


	
	
prevAll

	
get all Doms  before the Dom.

	


$("#test").prevAll()

	

	

		
		
	


	
	
next

	
get the index item after the Dom, index's default value is 0.

	


$("#test").next();

	

	

		
		
	


	
	
nextAll

	
get all Doms after the Dom.

	


$("#test").nextAll();

	

	

		
		
	


	
	
first

	
get the first child of the Dom.

	


$("#test").first();

	

	

		
		
	


	
	
last

	
get the last child of the Dom.

	


$("#test").last();

	

	

		
		
	


	
	
children

	
if param exists, return the index child of the Dom, or return all the children.

	


$("#test").children();
$("#test").children(2);

	

	

		
		
	


	
	
parent

	
get the parent element of the Dom, if param exists, then parent * index

	


$("#test").parent();
$("#test").parent(1); //get the parent's parent of the Dom

	

	

		
		
	


	
	
hasClass

	
whether the Dom include class.

	


$("#test").hasClass('classname');

	

	

		
		
	


	
	
addClass

	
add className to the Dom.

	


$("#test").addClass('classname');

	

	

		
		
	


	
	
removeClass

	
remove className.

	


$("#test").removeClass('classname');

	

	

		
		
	


	
	
removeAttr

	
remove attribute.

	


$("#test").removeAttr('id');

	

	

		
		
	


	
	
removeCSS

	
remove some keys in the style attribute of the Dom. if param not exist, remove all style attributes.
parameter like this: backgroundColor, marginTop

	


$("#test").removeCSS('display');

	

	

		
		
	


	
	
anim

	
perform a custom animation, please see Como.Anim

	


$("#test").anim().to('height', '50px').go();
