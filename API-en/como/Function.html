bind
apply scope, and can transfer some arguments
var test = function(){
	var a = [1, 2, 3, 4];
	a.each(function(i){
		...
	}.bind(this));
	//Como.Function.bind(function(i){}, this);
	//attention
	var test = Como.Function.bind(function(a, b, c){
		//the weight of binded paramters is more higher then function's own paramters.
		//eg. a=1, b=2, c=3
		//also bindEvent
	}, this, 1, 2);
	test(3);
}
