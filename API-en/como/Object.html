extend
Object.extend(target, src)
Copy all the key and value from an object to the target object.
var t = {
	key1: '111',
	key2: '222'
};
var s = {
	key1: '333',
	key3: function(){alert('444')}
};
Object.extend(t, s);
///t = {key1: '333', key2: '222', key3: function(){alert('444')}}
each
Object.each(obj, function(){})
Traverse an object and setup a function.
var t = {
	name: 'yanyun',
	age: 24
};
Object.each(t, function(val, key){
	alert('val = ' + val + '; key = ' + key);
});
clone
Object.clone(obj)
Clone an object totally, support String, Boolean, Date, Array, Function, Object and so on.
var t1 = "Hello";
var r1 = Object.clone(t1);
log(r1);
log(r1 == t1);

var t2 = 123;
var r2 = Object.clone(t2);
log(r2);
log(r2 == t2);

var t3 = true;
var r3 = Object.clone(t3);
log(r3);
log(r3 == t3);

var t4 = new Date();
var r4 = Object.clone(t4);
log(r4);
log(r4 == t4);

var t5 = [1,2,3,4,[5,6,7]];
var r5 = Object.clone(t5);
log(r5);
log(r5 == t5);

var t6 = function(){};
var r6 = Object.clone(t6);
log(r6);
log(r6 == t6);

var t7 = function(){};
t7.name = "kevin";
t7.prototype.set=function(){};
var r7 = Object.clone(t7);
r7.name="kevin1";
log(r7);
log(r7 == t7);

var t8 = {name:'kevin', desc: {age:25, sex: 'male'}, friends: ['h1', 'h2']};
var r8 = Object.clone(t8);
r8.name = 'kevin1';
log(r8);
log(r8 == t8);
