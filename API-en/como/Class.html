Class
This can realize a class of simple feature.
create inherit
create
Create a class. It must have the initialize function, also called constructor.
With Only mode, like this 'Como.Class.create(...);'
var test = Class.create({
	initialize: function(name){  //constructor
		this._name = name;	//prototype
	},
	getName: function(){  //function
		return this._name;
	}
});
var test1 = new test('yanyun'); //instance
test1.getName();
inherit
inherit a class, and not support multiple
var test = Class.inherit(baseClass, {
	initialize: function(name){  //constructor
		this.super_(name);  //this.super_: this function must to setup with the same parameters of parent, and then innerit.
		this._name = name;
	},
	getName: function(){  //function
		return this._name;
	}
});
var test1 = new test('yanyun'); //intance
test1.getName();
